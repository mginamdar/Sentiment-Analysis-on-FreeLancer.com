def get_content(url, pretend_agent=PRETEND_AGENT):
    global userCount
    userCount += 1
    if userCount % 15 == 0:
        print("User Fetch Count: ", userCount)

    """Gets the content from the URL using a pretend agent name"""
    return requests.get(url, headers={'User-Agent': pretend_agent}).content

    
def get_user_links(pagenum, pattern='/u/'):
    """Freelance user links from the main page that match the pattern"""
    content = get_content(FREELANCE_COM + C_SKILLS_USERS.format(pagenum))
    soup = BeautifulSoup(content, "lxml")
    # Links are from <a href="...", so we search for 'a' and get href if it exists (or '')
    all_links = [x.get('href') for x in soup.find_all('a')]
    # Get all links that match our pattern 
    user_links = [x for x in all_links if x and x.find('/u/') >= 0]
    return set(user_links)  # since there are duplicated

def get__user_details(user_profile):
    """Given a user_profile (web page text for that user), get the skills for the user"""
    
    global userCount
    userCount += 1
    
    profile = BeautifulSoup(user_profile, "lxml")
    
    #If city, country not available, skip the user
    try:
        earnings.append(float(profile.find('span', {'class': 'Earnings-label'}).text.strip()))
        recommendations.append(int(profile.find('span', {'data-target' : 'recommend-btn-val'}).text.strip()))
        
        ratingSoup = profile.find('span', {'Rating-total'})
        rating = ratingSoup.find(itemprop="ratingValue").get("content")
        numReviews.append(int(ratingSoup.find(itemprop="reviewCount").get("content")))
        
        if userCount % 100 == 0:
            print("User Process Count: ", userCount)

    except Exception as e:
        return None

import numpy as np
import pandas as pd
import matplotlib as mpl
import matplotlib.pyplot as plt
import statsmodels.api as sm


import seaborn as sns

#This cell will load excel data into dataframe and apply regression
#Regression charts are generated for each independent variable to see its impact on dependent variable

df = pd.read_excel('freelancer_1.xlsx', sheetname='Sheet1')

# Fit regression model (using the natural log of one of the regressors)
results = smf.ols('earnings ~ numReviews + recommendations', data=df).fit()
print(results.summary())

sns.set(color_codes=True)
sns.lmplot(x='recommendations', y='earnings', data=df);
sns.lmplot(x='numReviews', y='earnings', data=df);

fig = plt.figure(figsize=(12,8))
fig=sm.graphics.plot_regress_exog(results,'recommendations',fig=fig)
plt.show()

fig = plt.figure(figsize=(12,8))
fig=sm.graphics.plot_regress_exog(results,'numReviews',fig=fig)
plt.show()
